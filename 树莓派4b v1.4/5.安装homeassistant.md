# 安装home assistant

```shell
# docker安装home assistant Container
sudo docker run -d --name homeassistant --privileged --restart=unless-stopped -e TZ=TZ=Asia/Shanghai -v /data/homeassistant:/config --network=host ghcr.io/home-assistant/raspberrypi4-homeassistant:stable

# 安装home assistant Supervised
curl -Lo installer.sh https://raw.githubusercontent.com/home-assistant/supervised-installer/master/installer.sh
sudo bash installer.sh --machine MY_MACHINE
```

```shell
# MY_MACHINE 用以下中自己的设备名替换 
generic-x86-64
odroid-c2
odroid-n2
odroid-xu
qemuarm
qemuarm-64
qemux86
qemux86-64
raspberrypi
raspberrypi2

raspberrypi3-64
raspberrypi4-64
tinker
khadas-vim3
```
